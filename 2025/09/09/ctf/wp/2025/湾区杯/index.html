    def __init__(self, key):

        self.set_key(key)

    def set_key(self, key):

        if len(key) != 16:

            raise ValueError("Key must be 16 bytes long")

        key_bytes = key.encode('latin-1') if isinstance(key, str) else key

        k = self.str_to_int_array(key_bytes)

        # 扩展k数组以容纳32轮密钥

        k = k + [0] * 32  # 现在k有36个元素

        for i in range(4):

            k[i] ^= self.FK[i]

        for i in range(32):

            k[i+4] = k[i] ^ self.ckf(k[i+1], k[i+2], k[i+3], self.CK[i])

            self.sk[i] = k[i+4]

    def encrypt(self, plaintext):

        plaintext_bytes = plaintext.encode('latin-1') if isinstance(plaintext, str) else plaintext

        padding = 16 - (len(plaintext_bytes) % 16)

        plaintext_bytes += bytes([padding] * padding)

        ciphertext = b''

        for i in range(0, len(plaintext_bytes), 16):

            block = plaintext_bytes[i:i+16]

            ciphertext += self.crypt_block(block, self.ENCRYPT)

        return ciphertext

    def decrypt(self, ciphertext):

        ciphertext_bytes = ciphertext if isinstance(ciphertext, bytes) else base64.b64decode(ciphertext)

        plaintext = b''

        for i in range(0, len(ciphertext_bytes), 16):

            block = ciphertext_bytes[i:i+16]

            plaintext += self.crypt_block(block, self.DECRYPT)

        padding = plaintext[-1]

        if padding < 1 or padding > 16:

            return plaintext

        return plaintext[:-padding]

    def crypt_block(self, block, mode):

        x = self.str_to_int_array(block)

        if mode == self.ENCRYPT:

            for i in range(32):

                round_key = self.sk[i]

                x.append(x[0] ^ self.F(x[1], x[2], x[3], round_key))

                x.pop(0)

            x = x[::-1]

        else:

            for i in range(31, -1, -1):

                round_key = self.sk[i]

                x.append(x[0] ^ self.F(x[1], x[2], x[3], round_key))

                x.pop(0)

            x = x[::-1]

        return self.int_array_to_str(x)

    def F(self, x1, x2, x3, rk):

        return self.T(x1 ^ x2 ^ x3 ^ rk)

    def ckf(self, a, b, c, ck):

        return a ^ self.T(b ^ c ^ ck)

    def T(self, x):

        return self.L(self.S(x))

    def S(self, x):

        result = 0

        for i in range(4):

            byte = (x >> (24 - i * 8)) & 0xFF

            result |= (self.SboxTable[byte] << (24 - i * 8))

        return result

    def L(self, x):

        return x ^ self.rotl(x, 2) ^ self.rotl(x, 10) ^ self.rotl(x, 18) ^ self.rotl(x, 24)

    def rotl(self, x, n):

        return ((x << n) & 0xFFFFFFFF) | ((x >> (32 - n)) & 0xFFFFFFFF)

    def str_to_int_array(self, s):

class Quaternion:

    _slots_=("a","b","c","d")

    def __init__(self, a, b, c, d):

        self.p = p

        self.a = a % self.p

        self.b = b % self.p

        self.c = c % self.p

        self.d = d % self.p

    def __mul__(self, other):

        a1, b1, c1, d1 = self.a, self.b, self.c, self.d

        a2, b2, c2, d2 = other.a, other.b, other.c, other.d

        a_new = a1 * a2 - b1 * b2 - c1 * c2 - d1 * d2

        b_new = a1 * b2 + b1 * a2 + c1 * d2 - d1 * c2

        c_new = a1 * c2 - b1 * d2 + c1 * a2 + d1 * b2

        d_new = a1 * d2 + b1 * c2 - c1 * b2 + d1 * a2

        return Quaternion(a_new, b_new, c_new, d_new)

    def conj(self):

        return Quaternion(self.a, -self.b, -self.c, -self.d)

    def norm(self):

        return (self.a * self.a + self.b * self.b + self.c * self.c + self.d * self.d) % self.p

    def tup(self):

        return (self.a, self.b, self.c, self.d)
